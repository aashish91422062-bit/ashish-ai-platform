const canvas = new fabric.Canvas("canvas"); let imgObj = null; // LOAD IMAGE function loadImage(event) { const file = event.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(e) { fabric.Image.fromURL(e.target.result, function(img) { canvas.clear(); imgObj = img; img.set({ left: 100, top: 50, scaleX: 0.5, scaleY: 0.5 }); canvas.add(img); canvas.setActiveObject(img); canvas.renderAll(); }); }; reader.readAsDataURL(file); } // ROTATE LEFT function rotateLeft() { const obj = canvas.getActiveObject(); if (!obj) return; obj.rotate(obj.angle - 90); canvas.renderAll(); } // ROTATE RIGHT function rotateRight() { const obj = canvas.getActiveObject(); if (!obj) return; obj.rotate(obj.angle + 90); canvas.renderAll(); } // ZOOM IN function zoomIn() { canvas.setZoom(canvas.getZoom() + 0.1); } // ZOOM OUT function zoomOut() { canvas.setZoom(Math.max(0.2, canvas.getZoom() - 0.1)); } // RESET function resetImage() { canvas.clear(); canvas.setZoom(1); } // FILTERS function updateFilters() { const obj = canvas.getActiveObject(); if (!obj) return; const brightness = document.getElementById("brightness").value; const contrast = document.getElementById("contrast").value; const saturation = document.getElementById("saturation").value; obj.filters = [ new fabric.Image.filters.Brightness({ brightness: (brightness - 100) / 100 }), new fabric.Image.filters.Contrast({ contrast: (contrast - 100) / 100 }), new fabric.Image.filters.Saturation({ saturation: (saturation - 100) / 100 }) ]; obj.applyFilters(); canvas.renderAll(); } // DOWNLOAD (100% FIXED) function downloadImage() { canvas.discardActiveObject(); canvas.renderAll(); const dataURL = canvas.toDataURL({ format: "png", multiplier: 2 }); const link = document.createElement("a"); link.href = dataURL; link.download = "AI-Photo-Studio.png"; document.body.appendChild(link); link.click(); document.body.removeChild(link); }
