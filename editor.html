let canvas = new fabric.Canvas("canvas"); let currentImage = null; // IMAGE LOAD function loadImage(event) { const file = event.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(e) { fabric.Image.fromURL(e.target.result, function(img) { canvas.clear(); currentImage = img; img.set({ left: 100, top: 50, scaleX: 0.5, scaleY: 0.5 }); canvas.add(img); canvas.setActiveObject(img); canvas.renderAll(); }); }; reader.readAsDataURL(file); } // ROTATE LEFT function rotateLeft() { const obj = canvas.getActiveObject(); if (obj) { obj.rotate(obj.angle - 10); canvas.renderAll(); } } // ROTATE RIGHT function rotateRight() { const obj = canvas.getActiveObject(); if (obj) { obj.rotate(obj.angle + 10); canvas.renderAll(); } } // ZOOM IN function zoomIn() { const obj = canvas.getActiveObject(); if (obj) { obj.scaleX *= 1.1; obj.scaleY *= 1.1; canvas.renderAll(); } } // ZOOM OUT function zoomOut() { const obj = canvas.getActiveObject(); if (obj) { obj.scaleX *= 0.9; obj.scaleY *= 0.9; canvas.renderAll(); } } // RESET function resetImage() { canvas.clear(); } // DOWNLOAD IMAGE function downloadImage() { const dataURL = canvas.toDataURL({ format: "png", multiplier: 2 }); const link = document.createElement("a"); link.href = dataURL; link.download = "ai-photo-editor.png"; link.click(); } // FILTERS function updateFilters() { const obj = canvas.getActiveObject(); if (!obj) return; const brightness = document.getElementById("brightness").value; const contrast = document.getElementById("contrast").value; const saturation = document.getElementById("saturation").value; obj.filters = [ new fabric.Image.filters.Brightness({ brightness: (brightness - 100) / 100 }), new fabric.Image.filters.Contrast({ contrast: (contrast - 100) / 100 }), new fabric.Image.filters.Saturation({ saturation: (saturation - 100) / 100 }) ]; obj.applyFilters(); canvas.renderAll(); }
